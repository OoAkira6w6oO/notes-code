# 类模版

```C++
template <模板参数表>
class 类名
{
     数据成员
     成员函数
};


#include <iostream>
using namespace std;

// 类模板
template <typename T>//typename: 依赖类型，某个类型的定义依赖于模版参数
class Box {
private:
    T content;  // 成员类型由模板参数 T 决定
public:
    Box(T c) : content(c) {}  // 构造函数

    void setContent(T c) { content = c; }
    T getContent() const { return content; }

    void show() const { cout << content << endl; }
};

int main() {
    Box<int> intBox(123);      // T = int
    intBox.show();              // 输出 123

    Box<string> strBox("Hi");  // T = string
    strBox.show();              // 输出 Hi

    strBox.setContent("Hello");
    cout << strBox.getContent() << endl; // 输出 Hello
}

```
